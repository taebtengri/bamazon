use bamazon;

CREATE TABLE products (
 id INTEGER(11) AUTO_INCREMENT NOT NULL,

 product_name VARCHAR(30) NOT NULL,
  
 department_name VARCHAR(30) NOT NULL,

 price INTEGER(10),
 
 stock INTEGER(10),
  PRIMARY KEY (id)
);
